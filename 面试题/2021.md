### sleep的实现

function sleep(timer) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve()
        }, timer)
    })
}

### 防抖节流的区别
防抖： 一定周期内，如果重复触发，会重新计时
节流： 一定周期内，如果重复触发，不会执行
```js
// 防抖
function debounce(callback, timer) {
    let dtimer = null;
    return function() {
        if(dtimer) {
            clearTimeout(dtimer);
            dtimer = setTimeout(() => {
                callback();
            }, timer)
        }
        
    }
}

function throttle(callback, timer) {
    let flag = false;
    return function() {
        if(!flag) {
            flag = true;
            ttimer = setTimeout(() => {
               callback.call(this)
               flag = false;
           }, timer)
        }
        
    }
}

}, 1000))
```
